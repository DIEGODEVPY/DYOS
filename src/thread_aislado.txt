:DarArmaConMunicion {
        param1 - Actor
        param2 - Model ID
        param3 - Weapon ID
        param4 - Ammo
    }
    Model.Load(param2)
    repeat
        wait 0
    until Model.Available(param2)

    if param1 == ePara_Player
    then Actor.setGiveWeaponAndAmmo(_Player, param3, param4)
    else Actor.SetGiveWeaponAndAmmo($Actor($indActor,101i), param3, param4)
    end
    
    Model.Release(param2)
return
