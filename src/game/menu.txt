Menu.SetActiveItem($PANEL, 0)
:MenuDelJuego
Menu.SetColumn($PANEL, 0, 'M0000DT','M0000DU','M00008G','M0000FT','M0000ML',none,none,none,none,none,none,none,none)
FrenarRetorno()
FrenarSeleccion()
repeat
    wait 0
    if EntrarAlMenu()
    then
        if _Selected == 0 // Guardar
        then
            Game.ShowSaveScreen()
        end

        if _Selected == 1 // Hora
        then
            {$I time.txt}
            Menu.SetActiveItem($PANEL, 1)
            jump @MenuDelJuego
        end

        if _Selected == 2 // Clima
        then
            {$I weather.txt}
            Menu.SetActiveItem($PANEL, 2) 
            jump @MenuDelJuego
        end

        if _Selected == 3 // Tiempo
        then
            if $timeStoped == false
            then
                Enable()
                $timeStoped = true
                Weather.SetForceNow(0)
            else
                Disable()
                $timeStoped = false
            end
        end
        FrenarSeleccion()
    end
until SalirDelMenu()  

Menu.SetActiveItem($PANEL, menu_del_juego)
jump @MenuPrincipal
