Menu.SetActiveItem($PANEL, 0)
if $Seleccion == 5
then Text.PrintHelp('M0000IR')
end
:MenuDelJuego
Menu.SetColumn($PANEL, 0, 'M0000DT','M0000DU','M00008G','M0000FT','M0000IS','M0000IT','M0000IU','M0000IV',none,none,none,none,none)
FrenarRetorno()
FrenarSeleccion()
repeat
    wait 0
    if EntrarAlMenu()
    then
        if $Seleccion == 0 // Guardar
        then
            Game.ShowSaveScreen()
        end

        if $Seleccion == 1 // Tiempo
        then
            {$I time.txt}
            Menu.SetActiveItem($PANEL, 1)
            jump @MenuDelJuego
        end

        if $Seleccion == 2 // Clima
        then
            {$I weather.txt}
            Menu.SetActiveItem($PANEL, 2) 
            jump @MenuDelJuego
        end

        if $Seleccion == 3 // Nocturna
        then
            Display.SetThermalVision(false)
            Display.SetNightVision(true)
        end

        if $Seleccion == 4 // Termica
        then
            Display.SetNightVision(false)
            Display.SetThermalVision(true)
        end

        if $Seleccion == 5 // Despejada
        then
            if $cinematicMode == false
            then
                $cinematicMode = true
                Enable()
            else
                $cinematicMode = false
                Disable()
            end
        end

        if $Seleccion == 5 // Normal
        then 
            Display.SetThermalVision(false)
            Display.SetNightVision(false)
            Display.SetWidescreen(false)
        end

        if $Seleccion == 10 // velocidad
        then
            Menu.SetActiveItem($PANEL,0)
            //Menu.SetColumn($PANEL,0,'',)
            repeat
                wait 0
                Game.SetSpeed(1.0)
            until SalirDelMenu()
            Menu.SetActiveItem($PANEL, 10)
            jump @MenuDelJuego
        end
        FrenarSeleccion()
    end
until SalirDelMenu()  

Menu.SetActiveItem($PANEL, menu_del_juego)
jump @MenuPrincipal
