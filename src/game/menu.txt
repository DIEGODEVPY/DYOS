Menu.SetActiveItem($PANEL, 0)
:MenuDelJuego
Menu.SetColumn($PANEL, 0, 'M0000DT','M0000DU','M00008G','M0000FT',none,none,none,none,none,none,none,'M0000ML',none)
FrenarRetorno()
FrenarSeleccion()
repeat
    wait 0
    if EntrarAlMenu()
    then
        if $Seleccion == 0 // Guardar
        then
            Game.ShowSaveScreen()
        end

        if $Seleccion == 1 // Tiempo
        then
            {$I time.txt}
            Menu.SetActiveItem($PANEL, 1)
            jump @MenuDelJuego
        end

        if $Seleccion == 2 // Clima
        then
            {$I weather.txt}
            Menu.SetActiveItem($PANEL, 2) 
            jump @MenuDelJuego
        end

        if $Seleccion == 10 // velocidad
        then
            if $gameSpeed == 1.0
            then
                $gameSpeed = 0.0
            else
                $gameSpeed = 1.0
            end
            Game.SetSpeed($gameSpeed)
        end
        FrenarSeleccion()
    end
until SalirDelMenu()  

Menu.SetActiveItem($PANEL, menu_del_juego)
jump @MenuPrincipal
