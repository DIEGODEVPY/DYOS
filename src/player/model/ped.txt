FrenarSeleccion()
6@ = Actor.GetModel(_Player)
if 6@ == 290
then
	6@ = 0
	Player.SetModel(_Char, 6@)
end

repeat
	wait 0
	iluminar_al_jugador()

	if CloseDyos()
	then jump @CerrarElMenu
	end
	FixUpdate()

	if ParaDecrecer()
	then
		if $numActores < 50
		then
			$indActor = $numActores
			$numActores++
			$indActor++

			6@ = Actor.GetModel(_Player)
			GenerarActor()
			Aplicated()
			AparecerMenu()
			Animation.Remove("CLOTHES")
			ResetAnimPlayer()
			Camera.Restore()
			Menu.SetActiveItem($PANEL, 0)
			jump @MenuDeEdicionDelActor
		end
	end
	
	if ParaIzquierda()
	then
		6@ = Actor.GetModel(_Player)
		6@--
		
		ModeloAnterior_Actor()
			
		Model.Load(6@)
		Text.PrintHelpForever('M0000AW')
		wait 84
		repeat
			wait 0
			iluminar_al_jugador()
			FixUpdate()
		until Model.Available(6@)
		Text.ClearHelp()
		10@ = Actor.GetModel(_Player)
		Player.SetModel(_Char, 6@)
		iluminar_al_jugador()
		Model.Release(6@)
		Model.Release(10@)
		Text.PrintWithNumberNow('M000007', 6@, 1000, TextStyle.LightBlueTop)
		while ParaIzquierda() 
			wait 0
			iluminar_al_jugador()
			FixUpdate()
		end
	end
	if ParaDerecha()
	then
		6@ = Actor.GetModel(_Player)
		6@++

		ModeloSiguiente_Actor()
		
		Model.Load(6@)
		Text.PrintHelpForever('M0000AW')
		wait 84
		repeat
			wait 0
			iluminar_al_jugador()
			FixUpdate()
		until Model.Available(6@)
		Text.ClearHelp()
		10@ = Actor.GetModel(_Player)
		Player.SetModel(_Char, 6@)
		iluminar_al_jugador()

		Model.Release(6@)
		Model.Release(10@)
		Text.PrintWithNumberNow('M000007', 6@, 1000, TextStyle.LightBlueTop)
		while ParaDerecha()
			wait 0
			iluminar_al_jugador()
			FixUpdate()
		end
	end
	if ParaAtras()
	then
		6@ = Actor.GetModel(_Player)
		6@ -= 10
		
		ModeloAnterior_Actor()
			
		Model.Load(6@)
		Text.PrintHelpForever('M0000AW')
		wait 84
		repeat
			wait 0
			iluminar_al_jugador()
			FixUpdate()
		until Model.Available(6@)
		Text.ClearHelp()
		10@ = Actor.GetModel(_Player)
		Player.SetModel(_Char, 6@)  
		iluminar_al_jugador()
		Model.Release(6@)
		Model.Release(10@)
		Text.PrintWithNumberNow('M000007', 6@, 1000, TextStyle.LightBlueTop)
		while ParaAtras()
			wait 0
			iluminar_al_jugador()
			FixUpdate()
		end
	end
	if ParaAdelante()
	then
		6@ = Actor.GetModel(_Player)
		6@ += 10

		ModeloSiguiente_Actor()
		
		Model.Load(6@)
		Text.PrintHelpForever('M0000AW')
		wait 84
		repeat
			wait 0
			iluminar_al_jugador()
			FixUpdate()
		until Model.Available(6@)
		Text.ClearHelp()
		10@ = Actor.GetModel(_Player)
		Player.SetModel(_Char, 6@)
		iluminar_al_jugador()
		Model.Release(6@)
		Model.Release(10@)
		Text.PrintWithNumberNow('M000007', 6@, 1000, TextStyle.LightBlueTop)
		while ParaAdelante()
			wait 0
			iluminar_al_jugador()
			FixUpdate()
		end
	end
	if ParaEntrar()
	then
		while ParaEntrar()
			wait 0
			FixUpdate()
		end
		Aplicated()
		break
	end
until SalirDelMenu()
