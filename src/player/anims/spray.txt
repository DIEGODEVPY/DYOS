FrenarSeleccion()
Animation.Load('SPRAYCAN')
repeat
    wait 0
until Animation.Available('SPRAYCAN')
repeat
    wait 0
    if EntrarAlMenu()
    then
        if $colPlayer == 1
        then 
            if $Seleccion == 0
            then Task.PlayAnimSecondary(_Player, "SPRAYCAN_FIRE", 'SPRAYCAN', 4.0, true, false, false, false, -1)
            end
            if $Seleccion == 1
            then Task.PlayAnimSecondary(_Player, "SPRAYCAN_FULL", 'SPRAYCAN', 4.0, true, false, false, false, -1)
            end
        else
            if $Seleccion == 0
            then Task.PlayAnim(_Player, "SPRAYCAN_FIRE", 'SPRAYCAN', 4.0, true, false, false, false, -1)
            end
            if $Seleccion == 1
            then Task.PlayAnim(_Player, "SPRAYCAN_FULL", 'SPRAYCAN', 4.0, true, false, false, false, -1)
            end
        end
        FrenarSeleccion()
    end
until SalirDelMenu()
