:MovimientoVerticalDelMenu
	if ParaDerecha()
	then
		if not pagina_actual == numero_de_paginas
		then
			pagina_actual++ 
		end
	end
	if ParaIzquierda()
	then
		if not pagina_actual == 1
		then
			pagina_actual-- 
		end 
	end
	while ParaIzquierda()
		wait 0
		ImpresionDeLaPaginacion()
		Actor.Angle(_Player) = $tempvar_Angle
	end
	while ParaDerecha()
		wait 0
		ImpresionDeLaPaginacion()
		Actor.Angle(_Player) = $tempvar_Angle
	end
return

:MovimientoHorizontalDelMenu
	if $ModoDeCompilacion == ePara_PC
	then
		if $EmularAndroid == true
		then
			0@ = Menu.GetItemSelected($PANEL)
			if
				Key.IsPressed(VK.Num8)
			then
				0@--
				Menu.SetActiveItem($PANEL, 0@)
				while Key.IsPressed(VK.Num8)
					wait 0
				end
			end
			if
				Key.IsPressed(VK.K)
			then
				0@++
				Menu.SetActiveItem($PANEL, 0@)
				while Key.IsPressed(VK.K)
					wait 0
				end
			end
		end
	else
		0@ = Menu.GetItemSelected($PANEL)
		if 
			Widget.IsPressed(97) // Up arrow
		then
			0@--
			Menu.SetActiveItem($PANEL, 0@)
			while Widget.IsPressed(97) // Up arrow
				wait 0
			end
		end
		if
			Widget.IsPressed(98) // Down arrow
		then
			0@++
			Menu.SetActiveItem($PANEL, 0@)
			while Widget.IsPressed(98) // Down arrow
				wait 0
			end
		end
	end
return

:EntrarAlMenu
	ImpresionDeLaPaginacion()
	
	// Esta funcion se encarga del funcionamiento hibrido entre PC y Android
	if $Numero_De_Paginas > 1
	then
		MovimientoVerticalDelMenu()
	end

	MovimientoHorizontalDelMenu()
	
	$Seleccion = Menu.GetItemSelected($PANEL)
	
	:Selecionas
	if $ModoDeCompilacion == ePara_PC
	then
		if $EmularAndroid == true
		then
			if Key.IsPressed(VK.I)
			then return_true
			end
		else
			if Button.IsPressed(Pad.PLAYER1, VB.SPRINT)
			then return_true
			end
		end
	else
		if Widget.IsPressed(100) // Fire
		then return_true
		end
	end
	Actor.Angle(_Player) = $tempvar_Angle
return
