:MovimientoVerticalDelMenu
    if ParaDerecha()
    then
        if not pagina_actual == numero_de_paginas
        then
            pagina_actual++ 
        end
    end
    if ParaIzquierda()
    then
        if not pagina_actual == 1
        then
            pagina_actual-- 
        end 
    end
    while ParaIzquierda()
        wait 0
        ImpresionDeLaPaginacion()
    end
    while ParaDerecha()
        wait 0
        ImpresionDeLaPaginacion()
    end
return

:MovimientoHorizontalDelMenu
    if $ModoDeCompilacion == ePara_PC
    then
        if $EmularAndroid == true
        then
            0@ = Menu.GetItemSelected($PANEL)
            if
                Key.IsPressed(VK.Num8)
            then
                0@--
                Menu.SetActiveItem($PANEL, 0@)
                while Key.IsPressed(VK.Num8)
                    wait 0
                end
            end
            if
                Key.IsPressed(VK.K)
            then
                0@++
                Menu.SetActiveItem($PANEL, 0@)
                while Key.IsPressed(VK.K)
                    wait 0
                end
            end
        end
    else
        0@ = Menu.GetItemSelected($PANEL)
        if 
            Button.IsPressed(PAD.DISPLAY, 4) // Arriba
        then
            0@--
            Menu.SetActiveItem($PANEL, 0@)
            while Button.IsPressed(PAD.DISPLAY, 4) // Arriba
                wait 0
            end
        end
        if
            Button.IsPressed(PAD.DISPLAY, 6) // Abajo
        then
            0@++
            Menu.SetActiveItem($PANEL, 0@)
            while Button.IsPressed(PAD.DISPLAY, 6) // Abajo
                wait 0
            end
        end
    end
return

:EntrarAlMenu
    ImpresionDeLaPaginacion()
    
    // Esta funcion se encarga del funcionamiento hibrido entre PC y Android
    MovimientoVerticalDelMenu()
    MovimientoHorizontalDelMenu()
    
    $Seleccion = Menu.GetItemSelected($PANEL)
    
    :Selecionas

    then

        end
    end
