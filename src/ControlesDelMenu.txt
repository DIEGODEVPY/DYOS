		end

	then $despVelocity += 0.2
		if $despVelocity > 20.0
		then $despVelocity = 20.0
		end
	end
	int param1r = Math.FloatToInt_LsL($despVelocity)
	int a = 20
	a -= param1r
	Text.DisplayWithNumber(10.0, 105.0, 'M0000MT', a)

    0@ = Math.FloatToInt_LsG(x)

    0@ = Math.FloatToInt_LsG(y)

    0@ = Math.FloatToInt_LsG(z)
    Text.DisplayWithNumber(10.0, 95.0, 'M000013', 0@)

	float param5r = 0.0
	0078: param5r += frame_delta_time * 1.0 // (float)
	param5r /= $despVelocity
    if ParaDerecha()
	then 005B: x += param5r
	end
	if ParaIzquierda()
	then 0063: x -= param5r
	end
	if ParaAdelante()
	then 005B: y += param5r
	end
	if ParaAtras()
	then 0063: y -= param5r
	end
	if ParaArriba()
	then 005B: z += param5r
	end
	if ParaAbajo()

:GirarCamara
	if $ModoDeCompilacion == ePara_PC
	then
		if Key.IsPressed(1)
		then 
	        Player.SetCanMove(_Char,true)
			while Key.IsPressed(1)
				wait 0
				FixUpdate()
			end
			Player.SetCanMove(_Char,false)
		end
	else
		if Button.IsPressed(Pad.Display, 7)
		then
			Player.SetCanMove(_Char, true)
			while Button.IsPressed(Pad.Display, 7)
				wait 0
				FixUpdate()
			end
			Player.SetCanMove(_Char, false)
		end
	end

:MovimientoVerticalDelMenu
	if 803A: not $pagina_actual == $numero_de_paginas
	then
		if ParaDerecha()
		then $pagina_actual++ 
			while ParaDerecha()
				ImpresionDeLaPaginacion()
				wait 0
			end
		end
	end
	if not $pagina_actual == 1
	then
		if ParaIzquierda()
		then $pagina_actual-- 
			while ParaIzquierda()
				ImpresionDeLaPaginacion()
				wait 0
			end
		end
	end

:MovimientoHorizontalDelMenu
	if $ModoDeCompilacion == ePara_Android
	then
		0@ = Menu.GetItemSelected($PANEL)
		if 
			Widget.IsPressed(97) // Up arrow
		then
			0@--
			Menu.SetActiveItem($PANEL, 0@)
			while Widget.IsPressed(97) // Up arrow
				wait 0
				ImpresionDeLaPaginacion()
			end
		end
		if
			Widget.IsPressed(98) // Down arrow
		then
			0@++
			Menu.SetActiveItem($PANEL, 0@)
			while Widget.IsPressed(98) // Down arrow
				wait 0
				ImpresionDeLaPaginacion()
			end
		end
	end
return
