:CrearObjeto 
    FrenarSeleccion()
    $indObjeto = $numObjeto
    
    if $numObjeto < 101
    then
        $numObjeto++
        $indObjeto++
        
        GenerarObjeto()

        EsconderMenu()
        repeat
            wait 0
            if ParaIzquierda()
            then
                TIMERA = 0

                while ParaIzquierda()
                    wait 0
                    if TIMERA > 300
                    then
                        $ObjectModel -= 9
                        break
                    end
                end
                $ObjectModel--
                ModeloAnterior_Objeto()

                Object.Remove($Objeto($indObjeto,101i))
                GenerarObjeto()
                while ParaIzquierda()
                    wait 0
                end
            end
            if ParaDerecha()
            then
                TIMERA = 0

                while ParaDerecha()
                    wait 0
                    if TIMERA > 300
                    then
                        $ObjectModel += 9
                        break
                    end
                end
                $ObjectModel++
                ModeloSiguiente_Objeto()
                
                Object.Remove($Objeto($indObjeto,101i))
                GenerarObjeto()
                while ParaDerecha()
                    wait 0
                end
            end
            if ParaAtras()
            then
                $ObjectModel -= 10
                ModeloAnterior_Objeto()

                Object.Remove($Objeto($indObjeto,101i))
                GenerarObjeto()
                while ParaAtras()
                    wait 0
                end
            end

            if ParaAdelante()
            then
                $ObjectModel += 10
                ModeloSiguiente_Objeto()

                Object.Remove($Objeto($indObjeto,101i))
                GenerarObjeto()
                while ParaAdelante()
                    wait 0
                end
            end
            param1r = false
            if Selecionas()
            then
                param1r = true
                break
            end
            if ParaAbajo()
            then
                param1r = undefined
                break
            end
        until SalirDelMenu()

        AparecerMenu()
        Menu.SetActiveItem($PANEL, 0)
        Text.ClearHelp()
        Text.ClearPrints()
    else
        Text.PrintHelp('M00000M')
    end
return

:GenerarObjeto
    Model.Load($ObjectModel)
    Text.PrintHelpForever('M0000AW')
    Text.PrintWithNumberNow('M000007', $ObjectModel, 1000, 0)
    repeat
        wait 0
    until Model.Available($ObjectModel)
    Text.ClearHelp()
    
    $Objeto($indObjeto,101i) = Object.Create($ObjectModel, $tempvar_X_coord, $tempvar_Y_coord, $tempvar_Z_coord)
    Object.Angle($Objeto($indObjeto,101i)) = $tempvar_Angle
    Object.SetLinkInInterior($Objeto($indObjeto,101i), $Active_Interior)
    Object.SetCollisionDetection($Objeto($indObjeto,101i), false)  
    
    Model.Release($ObjectModel) 
    $colObjeto($indObjeto,101i) = false
    $visObject($indObjeto,101i) = true

    Text.PrintWithNumberNow('M000007', $ObjectModel, 1000, 0)
return
