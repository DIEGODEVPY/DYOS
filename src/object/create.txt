:CrearObjeto 
	FrenarSeleccion()
	$indObjeto = $numObjetos
	
	if $numObjetos < 101
	then
		$numObjetos++
		$indObjeto++
		
		GenerarObjeto()

		EsconderMenu()
		repeat
			wait 0
			FixUpdate()
			if ParaIzquierda()
			then
				$ObjectModel--
				ModeloAnterior_Objeto()
				FrenarIzquierda()
			end
			if ParaDerecha()
			then
				$ObjectModel++
				ModeloSiguiente_Objeto()
				FrenarDerecha()
			end
			if ParaAtras()
			then
				$ObjectModel -= 10
				ModeloAnterior_Objeto()
				FrenarAtras()
			end
			if ParaAdelante()
			then
				$ObjectModel += 10
				ModeloSiguiente_Objeto()
				FrenarAdelante()
			end
			if ParaDisminuir()
			then
				$ObjectModel -= 100
				ModeloAnterior_Objeto()
				FrenarDisminuir()
			end
			if ParaAumentar()
			then
				$ObjectModel += 100
				ModeloSiguiente_Objeto()
				FrenarAumentar()
			end

			param1r = false
			if Selecionas()
			then
				param1r = true
				break
			end
			if ParaAbajo()
			then
				param1r = undefined
				break
			end
		until SalirDelMenu()

		AparecerMenu()
		Menu.SetActiveItem($PANEL, 0)
		Text.ClearHelp()
		Text.ClearPrints()
	else
		Text.PrintHelp('M00000M')
	end
return

:GenerarObjeto
	Model.Load($ObjectModel)
	Text.PrintHelpForever('M0000AW')
	Text.PrintWithNumberNow('M000007', $ObjectModel, 1000, 0)
	repeat
		Wait0()
	until Model.Available($ObjectModel)
	Text.ClearHelp()
	
	GetUpdatePlayerPosition()
	$Objeto($indObjeto,101i) = Object.Create($ObjectModel, $tempvar_X_coord, $tempvar_Y_coord, $tempvar_Z_coord)
	Object.Angle($Objeto($indObjeto,101i)) = $tempvar_Angle
	Object.SetLinkInInterior($Objeto($indObjeto,101i), $Active_Interior)
	Object.SetCollisionDetection($Objeto($indObjeto,101i), false)
	Object.SetDestructible($Objeto($indObjeto,101i), false)

	Model.Release($ObjectModel) 

	$colObjeto($indObjeto,101i) = false
	$visObject($indObjeto,101i) = true
	$attObjeto($indObjeto,101i) = false

	Text.PrintWithNumberNow('M000007', $ObjectModel, 1000, 0)
return
