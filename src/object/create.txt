                end

                end
            end
            if ParaDerecha()
            then
                TIMERA = 0

                while ParaDerecha()
                    wait 0
                    if TIMERA > 300
                    then
                        $ObjectModel += 9
                        break
                    end
                end
                $ObjectModel++

                then $ObjectModel = 321
                end
                // IDs omitidos
                if and
                     $ObjectModel > 906
                     $ObjectModel < 910

                end
                
                Object.Remove($Objeto($indObjeto,101i))
                GenerarObjeto()
                while ParaDerecha()
                    wait 0
                end
            end
            if ParaAtras()
            then
                $ObjectModel -= 10

                then $ObjectModel = 18630
                end
                // IDs omitidos
                if and
                     $ObjectModel > 906
                     $ObjectModel < 910
                then $ObjectModel = 906

                     $ObjectModel > 906

:GenerarObjeto
    Model.Load($ObjectModel)
    Text.PrintHelpForever('M0000AW')
    repeat
        wait 0
    until Model.Available($ObjectModel)
    Text.ClearHelp()
    
    $Objeto($indObjeto,101i) = Object.Create($ObjectModel, $tempvar_X_coord, $tempvar_Y_coord, $tempvar_Z_coord)
    Object.Angle($Objeto($indObjeto,101i)) = $tempvar_Angle
    Object.SetLinkInInterior($Objeto($indObjeto,101i), $Active_Interior)
    Object.SetCollisionDetection($Objeto($indObjeto,101i), false)  
    
    Model.Release($ObjectModel) 
    $colObjeto($indObjeto,101i) = false
    $visObject($indObjeto,101i) = true

    Text.PrintWithNumberNow('M000007', $ObjectModel, 1000, 0)
return
