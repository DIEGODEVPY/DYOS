:GenerarActor
	Model.Load($ActorModel)
	Text.PrintHelpForever('M0000AW')
	if $ModoDeCompilacion == ePara_PC
	then
		TIMERB = 0
		while TIMERB <= 250
			Wait0()
		end
	else
		0@ = 0
		while 0@ <= 12
			Wait0()
			0@ += 1
		end
	end
	repeat
		Wait0()
	until Model.Available($ActorModel)
	Text.ClearHelp()

	GetUpdatePlayerPosition()
	$Actor($indActor,255i) = Actor.Create(PedType.CivMale, $ActorModel, $tempvar_X_coord, $tempvar_Y_coord, $tempvar_Z_coord)
	Actor.Angle($Actor($indActor,255i)) = $tempvar_Angle
	Actor.SetLinkInInterior($Actor($indActor,255i), $Active_Interior)
	
	Actor.SetKeepPosition($Actor($indActor,255i), true)
	Actor.SetCollisionDetection($Actor($indActor,255i), false)
	
	Model.Release($ActorModel)
	
	$statsActor($indActor,255i) = 0x0
	08C0: $statsActor($indActor,255i) set_false ACTOR_PROP_COLISION // bit 1 = false
	08BA: $statsActor($indActor,255i) set_true ACTOR_PROP_VISION // bit 1 = true
	08C0: $statsActor($indActor,255i) set_false ACTOR_PROP_IMMUNITION_BULLETS // bit 1 = false

	08C0: $statsActor($indActor,255i) set_false ACTOR_PROP_IMMUNITION_MELEE // bit 1 = false
	08C0: $statsActor($indActor,255i) set_false ACTOR_PROP_BLEEDING // bit 1 = false
