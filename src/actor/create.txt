:GenerarActor
    Model.Load($ActorModel)
    Text.PrintHelpForever('M0000AW')
    wait 124
    repeat
        wait 0
    until Model.Available($ActorModel)
    Text.ClearHelp()

    $Actor($indActor,101i) = Actor.Create(PedType.CivMale, $ActorModel, $tempvar_X_coord, $tempvar_Y_coord, $tempvar_Z_coord)
    Actor.Angle($Actor($indActor,101i)) = $tempvar_Angle
    Actor.SetLinkInInterior($Actor($indActor,101i), $Active_Interior)
    Actor.SetKeepPosition($Actor($indActor,101i), true)
    Actor.SetCollisionDetection($Actor($indActor,101i), false)
    
    Model.Release($ActorModel)
    
    $colActor($indActor,101i) = false 
    $visActor($indActor,101i) = true
    $indActor($indActor,101i) = false 

    Text.PrintWithNumberNow('M000007', $ActorModel, 1000, 0)
return
