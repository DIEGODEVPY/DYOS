Menu.SetActiveItem($PANEL, 0)
:MenuDeLaCamara
Menu.SetColumn($PANEL, 0, 'M0000IR','M0000IS','M0000IT','M0000IU','M0000IV',none,none,none,none,none,none,none,none)
FrenarRetorno()
FrenarSeleccion()
repeat
    wait 0
    if EntrarAlMenu()
    then
        if $Seleccion == 0 // Nocturna
        then
            Display.SetThermalVision(false)
            Display.SetNightVision(true)
        end

        if $Seleccion == 1 // Termica
        then
            Display.SetNightVision(false)
            Display.SetThermalVision(true)
        end

        if $Seleccion == 2 // Despejada
        then
            if $cinematicMode == false
            then
                $cinematicMode = true
                Enable()
            else
                $cinematicMode = false
                Disable()
            end
            
            if $ModoDeCompilacion == ePara_PC
            then
                Display.SetWidescreen($cinematicMode)
                Display.SetMenuAccessInWidescreen(true)
            else
                if $cinematicMode == true
                then
                    0DD9: write_mem_addr 2571800 value 0.0 size 4 add_ib 1 protect 1 
                    0DD9: write_mem_addr 2833896 value 0.0 size 4 add_ib 1 protect 1 
                else
                    0DD9: write_mem_addr 2571800 value 255.0 size 4 add_ib 1 protect 1 
                    0DD9: write_mem_addr 2833896 value 255.0 size 4 add_ib 1 protect 1
                end
            end
        end

        if $Seleccion == 3 // Normal
        then 
            Display.SetThermalVision(false)
            Display.SetNightVision(false)
            Display.SetWidescreen(false)
            $cinematicMode = false
            if $ModoDeCompilacion == ePara_Android
            then
                0DD9: write_mem_addr 2571800 value 255.0 size 4 add_ib 1 protect 1 
                0DD9: write_mem_addr 2833896 value 255.0 size 4 add_ib 1 protect 1
            end
        end
        FrenarSeleccion()
    end
until SalirDelMenu()  

Menu.SetActiveItem($PANEL, menu_de_la_camara)
jump @MenuPrincipal
